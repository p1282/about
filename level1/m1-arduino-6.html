버튼 + LED

버튼을 누르면 불이켜지고 안 누르면 불이 꺼진다

입력

없음

출력

없음

풀이

버튼 들어가는 선을 + 에 연결하고 나오는 선을 A0에 연결한다.
LED버튼은 7번핀과 연결한다.
            
int a;
void setup()
{
  pinMode(7, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  a = analogRead(A0);
  Serial.println(a);
  if (a == 1023) 
  {
    digitalWrite(7,HIGH);
    delay(1000);
  } else 
  {
    digitalWrite(7,LOW);
    delay(1000);
  }  
  delay(10);
}
-------         
            
버튼은 4번핀과 음극에 연결한다.
LED 는 8번핀에 연결한다.

void setup()
{
  pinMode(4, INPUT_PULLUP);
  pinMode(8, OUTPUT);  
  Serial.begin(9600);
}

void loop()
{
  int a = digitalRead(4);
  Serial.print("on off : ");
  Serial.print(a);
  if (a == 0) {
  	Serial.println("push OK ");
    digitalWrite(8,HIGH);
  } else if (a == 1) {
  	Serial.println("push NO");
    digitalWrite(8,LOW);  
  }
  delay(1000); 
}
