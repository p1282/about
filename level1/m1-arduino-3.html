LED 가변저항

가변저항을 돌려서 LED 불의 밝기를 조정한다. 저항값이 커지면 LED불이 밝아지고, 저항값이 작아지면 LED 불이 어두워진다.

입력

없음

출력

없음

풀이

int ledPin = 6;
void setup()
{
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  int a = analogRead(A0);
  Serial.println(a);
  int bright = map(a, 0, 1023, 0,255);
  analogWrite(ledPin, bright);
  delay(10); 
}
