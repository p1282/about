#-----
import turtle, random

c = int(input('거북이 몇마리 참가하나요? '))
col = ['blue','red','green','yellow','pink','cyan']
diff = 500 // c
turtles = []
for i in range(c):
    t = turtle.Turtle()
    t.shape('turtle')
    t.color(col[i%6])
    t.penup()
    x = -200
    y = i*diff - 230
    t.goto(x,y)
    t.pendown()
    turtles.append(t)
run = True
while run:    
    for j in range(c):
        step = random.randint(5,20)
        turtles[j].forward(step)
        x,y = turtles[j].pos()
        if x > 200:
            run = False
            break
