서보 Motor SG90

입력값이 들어오면 모터가 움직이고 LED불이 들어온다. 1번 숫자를 입력하면 모터를 180도 움직이면서 LED불이 들어오고, 2번 숫자를 입력하면 모터를 0도에 위치시키며 LED불이 꺼진다.

입력

없음

출력

없음

풀이

LED 는 8번 핀에 연결하고, 서보모터는 7번 핀에 연결한다.

#include 
Servo servo;

void setup()
{
  Serial.begin(9600);
  pinMode(8, OUTPUT);
  servo.attach(7);
}

void loop()
{
  if (Serial.available() > 0) {
    char a = Serial.read();
    Serial.println(a);
    if (a == '1') {
      digitalWrite(8,HIGH);
      servo.write(180);
    } else if (a == '2') {
      digitalWrite(8,LOW);
      servo.write(0);
    }
  }
  delay(100);
}
